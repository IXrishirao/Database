Table USERS {
  USER_ID bigint [pk]
  USER_NAME varchar(150)
  EMAIL varchar(150)
  PHONE varchar(20)
  USER_ADDRESS varchar(255)
  ROLE varchar(20) 
  CREATED_AT datetime
}

Table PROPERTIES {
  PROPERTY_ID bigint [pk]
  PROPERTY_NAME varchar(150)
  ADDRESS varchar(255)
  CITY varchar(100)
  STATE varchar(100)
  PROPERTY_SIZE_SQFT int
  BEDROOMS int
  BATHROOMS int
  OWNER_ID bigint
}

Table UTILITY_TYPES {
  UTILITY_TYPE_ID int [pk]
  UTILITY_NAME varchar(50)
}

Table METERS {
  METER_ID bigint [pk]
  PROPERTY_ID bigint [unique]
  UTILITY_TYPE_ID int
  INSTALLATION_DATE date
}

Table CONTRACTS {
  CONTRACT_ID bigint [pk]
  PROPERTY_ID bigint
  USER_ID bigint
  MOVE_IN_DATE date
  MOVE_OUT_DATE date
  SECURITY_DEPOSIT decimal
  MONTHLY_RENT decimal
  CONTRACT_DOCUMENT_URL varchar(255)
  SIGNED_DATE date
}

Table METER_READINGS {
  READING_ID bigint [pk]
  METER_ID bigint
  START_DATE date
  END_DATE date
  START_READING decimal
  END_READING decimal
}

Table TARIFF {
  TARIFF_ID bigint [pk]
  UTILITY_TYPE_ID int
  MIN_UNITS int
  MAX_UNITS int
  RATE_PER_UNIT decimal
}

Table INVOICES {
  INVOICE_ID bigint [pk]
  PROPERTY_ID bigint
  CONTRACT_ID bigint
  INVOICE_START date
  INVOICE_END date
  TOTAL_USAGE decimal
  TOTAL_AMOUNT decimal
  DUE_DATE date
  LATE_FINE decimal
  STATUS varchar(20)
}

Table INVOICE_DETAILS {
  DETAIL_ID bigint [pk]
  INVOICE_ID bigint
  DESCRIPTION varchar(200)
  UNITS decimal
  RATE decimal
  AMOUNT decimal
}

Table PAYMENTS {
  PAYMENT_ID bigint [pk]
  INVOICE_ID bigint
  USER_ID bigint
  PAYMENT_DATE date
  AMOUNT_PAID decimal
  PAYMENT_MODE varchar(50)
}

Ref: PROPERTIES.OWNER_ID > USERS.USER_ID

Ref: METERS.UTILITY_TYPE_ID > UTILITY_TYPES.UTILITY_TYPE_ID
Ref: CONTRACTS.PROPERTY_ID > PROPERTIES.PROPERTY_ID
Ref: CONTRACTS.USER_ID > USERS.USER_ID
Ref: METER_READINGS.METER_ID > METERS.METER_ID
Ref: TARIFF.UTILITY_TYPE_ID > UTILITY_TYPES.UTILITY_TYPE_ID
Ref: INVOICES.PROPERTY_ID > PROPERTIES.PROPERTY_ID
Ref: INVOICES.CONTRACT_ID > CONTRACTS.CONTRACT_ID
Ref: INVOICE_DETAILS.INVOICE_ID > INVOICES.INVOICE_ID
Ref: PAYMENTS.INVOICE_ID > INVOICES.INVOICE_ID
Ref: PAYMENTS.USER_ID > USERS.USER_ID
